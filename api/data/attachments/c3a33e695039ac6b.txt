pm.test('Status code 200', function () { pm.response.to.have.status(200); });
pm.test('Content-Type application/json', function () { pm.expect(pm.response.headers.get('Content-Type')).to.include('application/json'); });
pm.test('Body retorna bookingid e booking', function () {
  const json = pm.response.json();
  pm.expect(json).to.have.property('bookingid');
  pm.expect(json).to.have.property('booking');
  pm.expect(json.bookingid).to.be.a('number');
});
pm.test('Dados da reserva conferem com payload', function () {
  const json = pm.response.json();
  pm.expect(json.booking.firstname).to.eql('Tester');
  pm.expect(json.booking.lastname).to.eql('Antoiene');
  pm.expect(json.booking.totalprice).to.eql(111);
  pm.expect(json.booking.depositpaid).to.eql(true);
  pm.expect(json.booking.bookingdates.checkin).to.eql('2018-01-01');
  pm.expect(json.booking.bookingdates.checkout).to.eql('2019-01-01');
  pm.expect(json.booking.additionalneeds).to.eql('Breakfast');
});
