pm.test('Status code 200', function () { pm.response.to.have.status(200); });
pm.test('Tempo de resposta < 5s', function () { pm.expect(pm.response.responseTime).to.be.below(5000); });
pm.test('Body contem token', function () {
  const json = pm.response.json();
  pm.expect(json).to.have.property('token');
  pm.expect(json.token).to.be.a('string').and.not.empty;
});
