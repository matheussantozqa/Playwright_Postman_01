pm.test('Status code 200', function () { pm.response.to.have.status(200); });
pm.test('Content-Type application/json', function () { pm.expect(pm.response.headers.get('Content-Type')).to.include('application/json'); });
pm.test('Reserva retornada bate com a criada', function () {
  const json = pm.response.json();
  pm.expect(json).to.have.property('firstname', 'Tester');
  pm.expect(json).to.have.property('lastname', 'Antoiene');
  pm.expect(json).to.have.property('totalprice', 111);
  pm.expect(json).to.have.property('depositpaid', true);
  pm.expect(json.bookingdates).to.have.property('checkin', '2018-01-01');
  pm.expect(json.bookingdates).to.have.property('checkout', '2019-01-01');
});
